Шаблон определяет вывод в технологический журнал всех событий вместе со всеми свойствами. Объем выводимой информации при этом будет очень большим, однако, она может быть полезна при анализе сложных нештатных ситуаций. Данный шаблон рекомендуется использовать на этапе тестирования и при расследовании ошибок
<?xml version="1.0"?>
<config xmlns="http://v8.1c.ru/v8/tech-log">
	<system level="trace"/>
	<log history="4">
		<event>
			<ne property="name" value=""/>
		</event>
		<property name="all"/>
	</log>
	<dbmslocks/>
</config>
